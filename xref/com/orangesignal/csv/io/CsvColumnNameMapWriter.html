18  package com.orangesignal.csv.io;
19  
20  import java.io.Closeable;
21  import java.io.Flushable;
22  import java.io.IOException;
23  import java.util.ArrayList;
24  import java.util.Arrays;
25  import java.util.Collections;
26  import java.util.List;
27  import java.util.Map;
28  
29  import com.orangesignal.csv.CsvWriter;
30  import com.orangesignal.csv.filters.CsvNamedValueFilter;
31  
32  /**
33   * 項目名と項目値のマップで区切り文字形式データアクセスを行う区切り文字形式出力ストリームを提供します。
34   * 
35   * @author 杉澤 浩二
36   * @since 1.4.0
37   */
38  public class CsvColumnNameMapWriter implements Closeable, Flushable {
39  
40  	/**
41  	 * 区切り文字形式出力ストリームを保持します。
42  	 */
43  	private CsvWriter writer;
44  
45  	/**
46  	 * 項目名のリストを保持します。
47  	 */
48  	private List<String> columnNames;
49  
50  	/**
51  	 * 項目名の数を保存します。
52  	 */
53  	private int columnCount = -1;
54  
55  	/**
56  	 * 区切り文字形式データフィルタを保持します。
57  	 */
58  	private CsvNamedValueFilter filter;
59  
60  	// ------------------------------------------------------------------------
61  	// コンストラクタ
62  
63  	/**
64  	 * 指定された区切り文字形式出力ストリームを使用して、このクラスを構築するコンストラクタです。
65  	 * 
66  	 * @param writer 区切り文字形式出力ストリーム
67  	 * @throws IllegalArgumentException {@code writer} が {@code null} の場合。
68  	 */
69  	public CsvColumnNameMapWriter(final CsvWriter writer) {
70  		this(writer, null);
71  	}
72  
73  	/**
74  	 * 指定された区切り文字形式出力ストリームと項目名のリストを使用して、このクラスを構築するコンストラクタです。
75  	 * 
76  	 * @param writer 区切り文字形式出力ストリーム
77  	 * @param columnNames 項目名のリスト
78  	 * @throws IllegalArgumentException {@code writer} が {@code null} の場合。
79  	 */
80  	public CsvColumnNameMapWriter(final CsvWriter writer, final List<String> columnNames) {
81  		if (writer == null) {
82  			throw new IllegalArgumentException("CsvWriter must not be null");
83  		}
84  		this.writer = writer;
85  
86  		if (columnNames != null) {
87  			this.columnNames = Collections.unmodifiableList(columnNames);
88  		}
89  	}
90  
91  	// ------------------------------------------------------------------------
92  	// プライベート メソッド
93  
94  	/**
95  	 * Checks to make sure that the stream has not been closed
96  	 */
97  	private void ensureOpen() throws IOException {
98  		if (writer == null) {
99  			throw new IOException("CsvWriter closed");
100 		}
101 	}
102 
103 	private void ensureHeader(final Map<String, String> map) throws IOException {
104 		if (columnNames == null && map != null) {
105 			columnNames = new ArrayList<String>(map.keySet());
106 		}
107 		if (columnNames == null) {
108 			// ヘッダがない場合は例外をスローします。
109 			throw new IOException("No header is available");
110 		}
111 		if (columnCount == -1) {
112 			// ヘッダ部を処理します。
113 			writer.writeValues(columnNames);
114 			columnCount = columnNames.size();
115 		}
116 	}
117 
118 	// ------------------------------------------------------------------------
119 	// オーバーライド メソッド
120 
121 	@Override
122 	public void flush() throws IOException {
123 		synchronized (this) {
124 			ensureOpen();
125 			writer.flush();
126 		}
127 	}
128 
129 	@Override
130 	public void close() throws IOException {
131 		synchronized (this) {
132 			ensureOpen();
133 			writer.close();
134 			writer = null;
135 			columnNames = null;
136 			columnCount = -1;
137 		}
138 	}
139 
140 	// ------------------------------------------------------------------------
141 	// パブリック メソッド
142 
143 	/**
144 	 * 可能であれば項目名を書き込みます。項目名が既に書き込まれている場合、このメソッドは何も行いません。
145 	 * 
146 	 * @param map 項目名と項目値のマップ
147 	 * @throws IOException 入出力エラーが発生した場合
148 	 */
149 	public void writeHeader(final Map<String, String> map) throws IOException {
150 		synchronized (this) {
151 			ensureOpen();
152 			ensureHeader(map);
153 		}
154 	}
155 
156 	/**
157 	 * 指定された項目名と項目値のマップを書き込みます。項目名の書き込みが必要な場合は自動的に書き込みが行われます。
158 	 * 
159 	 * @param map 項目名と項目値のマップ
160 	 * @return 区切り文字形式データフィルタによって書き込みが行われなかった場合は {@code false} それ以外の場合は {@code true}
161 	 * @throws IOException 入出力エラーが発生した場合
162 	 */
163 	public boolean write(final Map<String, String> map) throws IOException {
164 		synchronized (this) {
165 			ensureOpen();
166 			ensureHeader(map);
167 
168 			// 要素が null の場合は null 出力します。
169 			if (map == null) {
170 				writer.writeValues(null);
171 				return true;
172 			}
173 
174 			final List<String> values = toValues(map);
175 			if (filter != null && !filter.accept(columnNames, values)) {
176 				return false;
177 			}
178 			writer.writeValues(values);
179 			return true;
180 		}
181 	}
182 
183 	private List<String> toValues(final Map<String, String> map) {
184 		final String[] values = new String[columnCount];
185 		for (int i = 0; i < columnCount; i++) {
186 			values[i] = map.get(columnNames.get(i));
187 		}
188 		return Arrays.asList(values);
189 	}
190 
191 	// ------------------------------------------------------------------------
192 	// セッター / ゲッター
193 
194 	/**
195 	 * 区切り文字形式データフィルタを返します。
196 	 * 
197 	 * @return 区切り文字形式データフィルタ。または {@code null}
198 	 */
199 	public CsvNamedValueFilter getFilter() {
200 		return filter;
201 	}
202 
203 	/**
204 	 * 区切り文字形式データフィルタを設定します。
205 	 * 
206 	 * @param filter 区切り文字形式データフィルタ
207 	 */
208 	public void setFilter(final CsvNamedValueFilter filter) {
209 		synchronized (this) {
210 			this.filter = filter;
211 		}
212 	}
213 
214 }
