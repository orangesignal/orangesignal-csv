17  package com.orangesignal.csv.manager;
18  
19  import java.text.Format;
20  
21  import com.orangesignal.csv.CsvConfig;
22  import com.orangesignal.csv.bean.CsvColumnNameMappingBeanTemplate;
23  import com.orangesignal.csv.filters.CsvNamedValueFilter;
24  import com.orangesignal.csv.handlers.ColumnNameMappingBeanListHandler;
25  
26  /**
27   * 区切り文字形式データの項目名を基準とする Java プログラム要素のリストと区切り文字形式データの統合入力インタフェースの実装クラスを提供します。
28   *
29   * @author Koji Sugisawa
30   * @since 1.2
31   */
32  public class CsvColumnNameMappingBeanLoader<T> extends AbstractCsvLoader<T, CsvColumnNameMappingBeanTemplate<T>, ColumnNameMappingBeanListHandler<T>, CsvColumnNameMappingBeanLoader<T>> {
33  
34  	/**
35  	 * コンストラクタです。
36  	 *
37  	 * @param cfg 区切り文字形式情報
38  	 * @param beanClass JavaBean の型
39  	 * @throws IllegalArgumentException パラメータが不正な場合
40  	 */
41  	protected CsvColumnNameMappingBeanLoader(final CsvConfig cfg, final Class<T> beanClass) {
42  		super(cfg, new ColumnNameMappingBeanListHandler<T>(beanClass));
43  	}
44  
45  	/**
46  	 * 指定された項目名と Java プログラム要素のフィールド名をマップへ追加します。
47  	 * 
48  	 * @param column 項目名
49  	 * @param field Java プログラム要素のフィールド名
50  	 * @return このオブジェクトへの参照
51  	 */
52  	public CsvColumnNameMappingBeanLoader<T> column(final String column, final String field) {
53  		getCsvListHandler().column(column, field);
54  		return this;
55  	}
56  
57  	/**
58  	 * 指定された項目名と Java プログラム要素のフィールド名をマップへ追加します。
59  	 * 
60  	 * @param column 項目名
61  	 * @param field Java プログラム要素のフィールド名
62  	 * @param format フィールドを処理するフォーマットオブジェクト (オプション)
63  	 * @return このオブジェクトへの参照
64  	 * @since 1.2
65  	 */
66  	public CsvColumnNameMappingBeanLoader<T> column(final String column, final String field, final Format format) {
67  		getCsvListHandler().column(column, field, format);
68  		return this;
69  	}
70  
71  	/**
72  	 * 区切り文字形式データフィルタを設定します。
73  	 * 
74  	 * @param filter 区切り文字形式データフィルタ
75  	 * @return このオブジェクトへの参照
76  	 * @since 1.2.3
77  	 */
78  	public CsvColumnNameMappingBeanLoader<T> filter(final CsvNamedValueFilter filter) {
79  		getCsvListHandler().filter(filter);
80  		return this;
81  	}
82  
83  }
