17  package com.orangesignal.csv;
18  
19  import java.io.Serializable;
20  
21  /**
22   * {@link CsvToken} を実装したデフォルトの実装クラスを提供します。
23   *
24   * @author Koji Sugisawa
25   */
26  public class SimpleCsvToken implements Serializable, CsvToken {
27  
28  	private static final long serialVersionUID = 9180143360016401191L;
29  
30  	/**
31  	 * トークンの値を保持します。
32  	 */
33  	private String value;
34  
35  	/**
36  	 * トークンの開始物理行番号を保持します。
37  	 */
38  	private int startLineNumber;
39  
40  	/**
41  	 * トークンの終了物理行番号を保持します。
42  	 */
43  	private int endLineNumber;
44  
45  	/**
46  	 * トークンが囲み文字で囲まれていたかどうかを保持します。
47  	 */
48  	private boolean enclosed;
49  
50  	/**
51  	 * デフォルトコンストラクタです。
52  	 */
53  	protected SimpleCsvToken() {}
54  
55  	/**
56  	 * コンストラクタです。
57  	 *
58  	 * @param value トークンの値
59  	 * @param start トークンの開始物理行番号
60  	 * @param end トークンの終了物理行番号
61  	 * @param enclosed トークンが囲み文字で囲まれていたかどうか
62  	 */
63  	public SimpleCsvToken(final String value, final int start, final int end, final boolean enclosed) {
64  		this.value = value;
65  		this.startLineNumber = start;
66  		this.endLineNumber = end;
67  		this.enclosed = enclosed;
68  	}
69  
70  	@Override public String getValue() { return value; }
71  	@Override public int getStartLineNumber() { return startLineNumber; }
72  	@Override public int getEndLineNumber() { return endLineNumber; }
73  	@Override public boolean isEnclosed() { return enclosed; }
74  
75  }
