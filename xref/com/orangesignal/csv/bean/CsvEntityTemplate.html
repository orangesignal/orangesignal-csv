41  /**
42   * 区切り文字形式データ注釈要素 {@link CsvEntity} で注釈付けされた Java プログラム要素の操作を簡素化するヘルパークラスを提供します。
43   * 
44   * @author Koji Sugisawa
45   * @since 1.4.0
46   * @see CsvEntity
47   * @see CsvColumn
48   * @see CsvColumns
49   */
50  public class CsvEntityTemplate<T> extends AbstractCsvBeanTemplate<T, CsvEntityTemplate<T>> implements CsvEntityOperation<CsvEntityTemplate<T>> {

62  	 * 
63  	 * @param entityClass 区切り文字形式データ注釈要素 {@link CsvEntity} で注釈付けされた Java プログラム要素の型
64  	 * @return 新しい {@link CsvEntityTemplate} のインスタンス
65  	 * @throws IllegalArgumentException {@code entityClass} が {@code null} または不正な場合
66  	 */
67  	public static <T> CsvEntityTemplate<T> newInstance(final Class<T> entityClass) {
68  		return new CsvEntityTemplate<T>(entityClass);

78  	 * @throws IllegalArgumentException {@code entityClass} が {@code null} または不正な場合
79  	 */
80  	public CsvEntityTemplate(final Class<T> entityClass) {
81  		super(entityClass);
82  		if (entityClass.getAnnotation(CsvEntity.class) == null) {
83  			throw new IllegalArgumentException(String.format("No CsvEntity is available %s", entityClass.getName()));

99  	/**
100 	 * 指定された区切り文字形式データの値リストが含まれる必要があるかどうかを判定します。
101 	 * 
102 	 * @param columnNames 区切り文字形式データの項目名リスト
103 	 * @param values 区切り文字形式データの項目値のリスト
104 	 * @return {@code values} が含まれる必要がある場合は {@code true}
105 	 */
106 	public boolean isAccept(final List<String> columnNames, final List<String> values) {
107 		return (filter != null && !filter.accept(columnNames, values));
108 	}

110 	/**
111 	 * 項目名のリストを作成して返します。
112 	 * 
113 	 * @return 項目名のリスト
114 	 */
115 	public List<String> createColumnNames() {
116 		final SortedMap<Integer, String> positionMap = new TreeMap<Integer, String>();
117 		final List<String> adding = new ArrayList<String>();
118 
119 		for (final Field f : getType().getDeclaredFields()) {
120 			final CsvColumns columns = f.getAnnotation(CsvColumns.class);
121 			if (columns != null) {
122 				for (final CsvColumn column : columns.value()) {
123 					final int pos = column.position();
124 					final String name = defaultIfEmpty(column.name(), f.getName());
125 					if (pos >= 0) {
126 						if (positionMap.containsKey(pos)) {
127 							continue;
128 						}
129 						positionMap.put(pos, name);
130 					} else {
131 						adding.add(name);
132 					}
133 				}
134 			}
135 			final CsvColumn column = f.getAnnotation(CsvColumn.class);
136 			if (column != null) {
137 				final int pos = column.position();
138 				final String name = defaultIfEmpty(column.name(), f.getName());
139 				if (pos >= 0) {
140 					if (positionMap.containsKey(pos)) {
141 						continue;
142 					}
143 					positionMap.put(pos, name);
144 				} else {
145 					adding.add(name);
146 				}
147 			}
148 		}
149 
150 		final int max = positionMap.size() > 0 ? positionMap.lastKey().intValue() + 1 : 0;
151 		final String[] names = new String[max];
152 		for (final Map.Entry<Integer, String> entry : positionMap.entrySet()) {
153 			names[entry.getKey().intValue()] = entry.getValue();
154 		}
155 
156 		final List<String> results = new ArrayList<String>(Arrays.asList(names));
157 		if (adding.size() > 0) {
158 			results.addAll(adding);
159 		}
160 		return results;
161 	}

163 	public void prepare(final List<String> names, final Field[] fields) {
164 		super.valueParserMapping(new HashMap<String, Format>(0));
165 		super.valueFormatterMapping(new HashMap<Object, Format>(0));
166 
167 		// 書式オブジェクトの準備を行います。
168 		for (final Field f : fields) {
169 			final CsvColumns columns = f.getAnnotation(CsvColumns.class);
170 			if (columns != null) {
171 				for (final CsvColumn column : columns.value()) {
172 					final Format format = createFormat(column, f);
173 					if (format != null) {
174 						setValueParser(f.getName(), format);
175 						setValueFormatter(getPosition(column, f, names), format);
176 					}
177 				}
178 			}
179 			final CsvColumn column = f.getAnnotation(CsvColumn.class);
180 			if (column != null) {
181 				final Format format = createFormat(column, f);
182 				if (format != null) {
183 					setValueParser(f.getName(), format);
184 					setValueFormatter(getPosition(column, f, names), format);
185 				}
186 			}
187 		}
188 	}

192 	public static int getPosition(final CsvColumn column, final Field f, final List<String> names) {
193 		// 項目位置が指定されている場合は、項目位置から値を取得します。
194 		int pos = column.position();
195 		// 項目位置が指定されておらずヘッダ行が使用可能な場合は項目名から値を取得します。
196 		if (pos < 0 && names != null) {
197 			pos = names.indexOf(defaultIfEmpty(column.name(), f.getName()));
198 		}
199 		// 項目名と項目位置のどちらも使用できない場合は例外をスローします。
200 		if (pos == -1) {
201 			throw new IllegalStateException(String.format("Invalid CsvColumn field %s", f.getName()));
202 		}
203 		return pos;
204 	}

206 	public static String defaultIfEmpty(final String str, final String defaultStr) {
207 		return (str == null || str.isEmpty()) ? defaultStr : str;
208 	}

212 	private static Format createFormat(final CsvColumn column, final Field f) {
213 		final String pattern = column.format();
214 		if (pattern.isEmpty()) {
215 			return null;
216 		}
217 
218 		final Locale locale = column.language().isEmpty() ? Locale.getDefault() : new Locale(column.language(), column.country());
219 		if (Date.class.isAssignableFrom(f.getType())) {
220 			final SimpleDateFormat format = new SimpleDateFormat(pattern, locale);
221 			if (!column.timezone().isEmpty()) {
222 				format.setTimeZone(TimeZone.getTimeZone(column.timezone()));
223 			}
224 			return format;
225 		}
226 		final DecimalFormat format = new DecimalFormat(pattern, DecimalFormatSymbols.getInstance(locale));
227 		if (!column.currency().isEmpty()) {
228 			format.setCurrency(Currency.getInstance(column.currency()));
229 		}
230 		return format;
231 	}
